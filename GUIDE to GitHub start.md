# 1. Инициализировать Git
git init

# 2. Добавить все файлы
git add .

# 3. Сделать первый commit
git commit -m "Initial commit"

# 4. Назвать главную ветку main
git branch -M main

# 5. Подключить GitHub-репозиторий
git remote add origin https://github.com/sock-puppet-dev/HTMLACSS.git

# 6. Отправить проект на GitHub
git push -u origin main
